# Estimation
# Calculates estimates of mean cost given censored data Use ccmean (ccostr) With (In) R Software
# Estimates of mean cost with censored data Use ccmean (ccostr) With (In) R Software
ccmean <- ccmean(ccmean, L = 1000)
